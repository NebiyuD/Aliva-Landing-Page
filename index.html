import { FHIRClient } from '@aliva/fhir';
import { AIEngine } from '@aliva/intelligence';

const patient = await FHIRClient.getPatient(id);
const insights = await AIEngine.analyze({
  data: patient.timeline,
  context: 'emergency_department',
  normalize: true
});

// Real-time conflict detection
const conflicts = await insights.detectConflicts();
if (conflicts.critical) {
  alert.notify(provider, conflicts);
}
